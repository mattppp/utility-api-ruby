def to_json(opts = {})
  hash_processed = Hash[
    to_h.
      reject { |_, v| v.nil? }.
      map { |k, v| [k, v.is_a?(Time) ? v.iso8601 : v] }
  ]
  hash_processed.to_json(opts)
end
