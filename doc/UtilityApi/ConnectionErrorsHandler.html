def on_complete(env)
  case env[:status]
  when 400
    raise Errors::BadRequest
  when 404
    raise Errors::NotFound
  when 500
    raise Errors::InternalServerError
  when 400..600
    raise Errors::BaseError, "HTTP code: #{env[:status]}"
  else
    # do nothing, as the status code represents successful response
  end
end
